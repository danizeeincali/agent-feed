// Export utilities for session persistence
export const claudeSessionsPersistence = {
  async syncSession(session: ClaudeSession): Promise<void> {
    try {
      const existingSession = await ClaudeSessionsDB.getSession(session.id);
      
      if (existingSession) {
        await ClaudeSessionsDB.updateSession(session.id, session);
      } else {
        await ClaudeSessionsDB.createSession(session);
      }
    } catch (error) {
      logger.error('Failed to sync Claude session:', error);
    }
  },
 
  async syncAgent(agent: ClaudeAgent, sessionId: string): Promise<void> {

    }
  },
