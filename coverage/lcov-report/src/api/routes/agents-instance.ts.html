const getCategoryColor = (category: string) => {
  const colors: Record<string, string> = {
    coordination: '#3B82F6',
    productivity: '#10B981',
    analysis: '#F59E0B',
    documentation: '#8B5CF6',
    research: '#EC4899',
    meta: '#6B7280',
    social: '#06B6D4',
    content: '#84CC16'
  };
  return colors[category] || '#6B7280';
};
