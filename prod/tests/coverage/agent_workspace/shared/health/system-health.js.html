    /**
     * Perform comprehensive health check
     */
    async performHealthCheck(type = 'quick') {
        const startTime = Date.now();
        const healthCheck = {
            timestamp: new Date(),
            type, // 'quick' or 'deep'
            overall: {
                status: 'healthy',
                score: 100,
                message: 'All systems operational'
            },
            components: {},
            metrics: {
                totalComponents: 0,
                healthyComponents: 0,
                degradedComponents: 0,
                unhealthyComponents: 0,
                criticalComponents: 0
            },
            performance: {
                checkDuration: 0,
                lastCheck: this.lastHealthCheck,
                checksToday: this.getChecksToday()
            },
            recommendations: []
        };
 
        try {
            // Perform component health checks
            await this.checkPostingIntelligenceHealth(healthCheck);
            await this.checkDatabaseHealth(healthCheck);
            await this.checkRedisCacheHealth(healthCheck);
            await this.checkAPIGatewayHealth(healthCheck);
            await this.checkFeedIntelligenceHealth(healthCheck);
            await this.checkMonitoringHealth(healthCheck);
            await this.checkAgentIntegrationsHealth(healthCheck);
 
            // Perform system-level checks
            if (type === 'deep') {
                await this.performDeepHealthChecks(healthCheck);
            }
 
            // Calculate overall health
            this.calculateOverallHealth(healthCheck);
 
            // Generate recommendations
            this.generateHealthRecommendations(healthCheck);
 
            // Update metrics
            this.updateHealthMetrics(healthCheck);
 
        } catch (error) {
            healthCheck.overall = {
                status: 'critical',
                score: 0,
                message: `Health check failed: ${error.message}`,
                error: error.stack
            };
        }
 
        healthCheck.performance.checkDuration = Date.now() - startTime;
        this.lastHealthCheck = healthCheck;
        
        // Add to history
        this.addToHistory(healthCheck);
 
        return healthCheck;
    }

    /**
     * Check agent integrations health
     */
    async checkAgentIntegrationsHealth(healthCheck) {
        const componentName = 'agent-integrations';
        const integrationResults = {};
 
        for (const [agentType, integration] of this.components[componentName]) {
            const startTime = Date.now();
            
            try {
                const health = await Promise.race([
                    integration.healthCheck(),
                    this.timeoutPromise(this.config.componentTimeout, `${agentType} integration timeout`)
                ]);
 
                const responseTime = Date.now() - startTime;
                const status = this.evaluateComponentHealth(health, responseTime);
 
                integrationResults[agentType] = {
                    status: status,
                    responseTime,
                    details: health,
                    lastCheck: new Date()
                };
 
            } catch (error) {
                integrationResults[agentType] = {
                    status: 'unhealthy',
                    responseTime: Date.now() - startTime,
                    error: error.message,
                    lastCheck: new Date()
                };
            }
        }
 
        // Calculate overall agent integrations health
        const statuses = Object.values(integrationResults).map(r => r.status);
        const healthyCount = statuses.filter(s => s === 'healthy').length;
        const totalCount = statuses.length;
        
        let overallStatus = 'healthy';
        if (totalCount === 0) {
            overallStatus = 'unknown';
        } else if (healthyCount === totalCount) {
            overallStatus = 'healthy';
        } else if (healthyCount >= totalCount * 0.7) {
            overallStatus = 'degraded';
        } else {
            overallStatus = 'unhealthy';
        }
 
        healthCheck.components[componentName] = {
            status: overallStatus,
            healthyIntegrations: healthyCount,
            totalIntegrations: totalCount,
            integrations: integrationResults,
            lastCheck: new Date()
        };
    }

    /**

        const totalMem = os.totalmem();
        const freeMem = os.freemem();

        const memPercent = ((totalMem - freeMem) / totalMem) * 100;
 
        return {

            },
            memory: {
                usage: memPercent,
                heap: {
                    used: memUsage.heapUsed,
                    total: memUsage.heapTotal,
                    percent: (memUsage.heapUsed / memUsage.heapTotal) * 100
                },
                status: this.evaluateThreshold(memPercent, this.config.thresholds.memory)
            },
            uptime: process.uptime(),
            loadAverage: os.loadavg()
        };
    }

    /**
     * Evaluate component health based on response and timing
     */
    evaluateComponentHealth(health, responseTime) {
        // If the health check itself failed
        if (!health || typeof health !== 'object') {
            return 'critical';
        }
 
        // Check explicit health status
        if (health.status) {
            const status = health.status.toLowerCase();
            if (['critical', 'failed', 'error'].includes(status)) return 'critical';
            if (['unhealthy', 'down'].includes(status)) return 'unhealthy';
            if (['degraded', 'warning'].includes(status)) return 'degraded';
            if (['healthy', 'ok', 'up'].includes(status)) {
                // Still check response time
                return this.evaluateResponseTime(responseTime);
            }
        }
 
        // Default evaluation based on response time
        return this.evaluateResponseTime(responseTime);
    }
 
    /**
     * Evaluate response time against thresholds
     */
    evaluateResponseTime(responseTime) {
        const thresholds = this.config.thresholds.responseTime;
        
        if (responseTime <= thresholds.good) return 'healthy';
        if (responseTime <= thresholds.acceptable) return 'healthy'; // Still healthy but slower
        if (responseTime <= thresholds.poor) return 'degraded';
        return 'unhealthy'; // > 1000ms
    }
 
    /**
     * Evaluate metric against thresholds
     */
    evaluateThreshold(value, thresholds) {
        if (value <= thresholds.good) return 'healthy';
        if (value <= thresholds.acceptable) return 'healthy';
        if (value <= thresholds.poor) return 'degraded';
        return 'critical';
    }
 
    /**
     * Calculate overall system health
     */
    calculateOverallHealth(healthCheck) {
        const components = healthCheck.components;
        const weights = this.config.componentWeights;
        
        let totalScore = 0;
        let totalWeight = 0;
        let criticalCount = 0;
        let unhealthyCount = 0;
        let degradedCount = 0;
        let healthyCount = 0;
 
        for (const [componentName, component] of Object.entries(components)) {
            const weight = weights[componentName] || 0.05; // Default weight
            let componentScore = 0;
 
            // Handle agent-integrations differently (it has sub-components)
            if (componentName === 'agent-integrations' && component.integrations) {
                const statuses = Object.values(component.integrations).map(i => i.status);
                const avgScore = this.calculateAverageStatusScore(statuses);
                componentScore = avgScore;
            } else {
                componentScore = this.statusToScore(component.status);
            }
 
            totalScore += componentScore * weight;
            totalWeight += weight;
 
            // Count components by status
            const status = component.status;
            if (status === 'critical') criticalCount++;
            else if (status === 'unhealthy') unhealthyCount++;
            else if (status === 'degraded') degradedCount++;
            else if (status === 'healthy') healthyCount++;
        }
 
        // Calculate weighted average score
        const averageScore = totalWeight > 0 ? (totalScore / totalWeight) : 0;
 
        // Update metrics
        healthCheck.metrics.totalComponents = Object.keys(components).length;
        healthCheck.metrics.healthyComponents = healthyCount;
        healthCheck.metrics.degradedComponents = degradedCount;
        healthCheck.metrics.unhealthyComponents = unhealthyCount;
        healthCheck.metrics.criticalComponents = criticalCount;
 
        // Determine overall status
        let overallStatus = 'healthy';
        let message = 'All systems operational';
 
        if (criticalCount > 0) {
            overallStatus = 'critical';
            message = `${criticalCount} critical component(s) detected`;
        } else if (unhealthyCount > 0) {
            overallStatus = 'unhealthy';
            message = `${unhealthyCount} unhealthy component(s) detected`;
        } else if (degradedCount > 0) {
            overallStatus = 'degraded';
            message = `${degradedCount} component(s) experiencing degraded performance`;
        } else if (averageScore < 90) {
            overallStatus = 'degraded';
            message = 'System performance below optimal levels';
        }
 
        healthCheck.overall = {
            status: overallStatus,
            score: Math.round(averageScore),
            message,
            breakdown: {
                healthy: healthyCount,
                degraded: degradedCount,
                unhealthy: unhealthyCount,
                critical: criticalCount
            }
        };
    }
 
    /**
     * Calculate average score from status array
     */
    calculateAverageStatusScore(statuses) {
        if (statuses.length === 0) return 0;
        
        const total = statuses.reduce((sum, status) => sum + this.statusToScore(status), 0);
        return total / statuses.length;
    }
 
    /**
     * Convert status to numeric score
     */
    statusToScore(status) {
        const scores = {
            'healthy': 100,
            'degraded': 75,
            'unhealthy': 50,
            'critical': 0,
            'unknown': 25
        };
        
        return scores[status] || scores.unknown;
    }
 
    /**
     * Generate health recommendations
     */
    generateHealthRecommendations(healthCheck) {
        const recommendations = [];
 
        // Check for critical components
        for (const [componentName, component] of Object.entries(healthCheck.components)) {
            if (component.status === 'critical') {
                recommendations.push({
                    severity: 'critical',
                    component: componentName,
                    issue: 'Component is in critical state',
                    recommendation: `Immediate attention required for ${componentName}`,
                    action: 'restart_component'
                });
            } else if (component.status === 'unhealthy') {
                recommendations.push({
                    severity: 'high',
                    component: componentName,
                    issue: 'Component is unhealthy',
                    recommendation: `Investigate and resolve issues with ${componentName}`,
                    action: 'investigate_component'
                });
            } else if (component.responseTime && component.responseTime > this.config.thresholds.responseTime.poor) {
                recommendations.push({
                    severity: 'medium',
                    component: componentName,
                    issue: 'Slow response time detected',
                    recommendation: `Optimize performance for ${componentName}`,
                    action: 'optimize_performance'
                });
            }
        }
 
        // Check system-wide patterns
        if (healthCheck.metrics.degradedComponents > healthCheck.metrics.healthyComponents) {
            recommendations.push({
                severity: 'high',
                component: 'system',
                issue: 'Majority of components are degraded',
                recommendation: 'System-wide performance review needed',
                action: 'system_review'
            });
        }
 
        // Check for deep check issues if available
        if (healthCheck.deepChecks) {
            const resources = healthCheck.deepChecks.systemResources;
            if (resources) {
                if (resources.cpu.status === 'critical') {
                    recommendations.push({
                        severity: 'high',
                        component: 'system',
                        issue: 'High CPU usage detected',
                        recommendation: 'Scale up resources or optimize CPU-intensive operations',
                        action: 'scale_resources'
                    });
                }
                
                if (resources.memory.status === 'critical') {
                    recommendations.push({
                        severity: 'high',
                        component: 'system',
                        issue: 'High memory usage detected',
                        recommendation: 'Investigate memory leaks or scale memory resources',
                        action: 'investigate_memory'
                    });
                }
            }
        }
 
        healthCheck.recommendations = recommendations;
    }

    /**
     * Get health trends
     */
    getHealthTrends(hours = 24) {
        const cutoffTime = new Date(Date.now() - (hours * 60 * 60 * 1000));
        const recentChecks = this.healthHistory.filter(check => 
            new Date(check.timestamp) > cutoffTime
        );
 
        if (recentChecks.length === 0) {
            return { trend: 'unknown', checks: 0 };
        }
 
        const scores = recentChecks.map(check => check.overall.score);
        const averageScore = scores.reduce((sum, score) => sum + score, 0) / scores.length;
        
        let trend = 'stable';
        if (recentChecks.length > 1) {
            const firstHalf = scores.slice(0, Math.floor(scores.length / 2));
            const secondHalf = scores.slice(Math.floor(scores.length / 2));
            
            const firstAvg = firstHalf.reduce((sum, score) => sum + score, 0) / firstHalf.length;
            const secondAvg = secondHalf.reduce((sum, score) => sum + score, 0) / secondHalf.length;
            
            const difference = secondAvg - firstAvg;
            
            if (difference > 5) trend = 'improving';
            else if (difference < -5) trend = 'declining';
        }
 
        return {
            trend,
            averageScore,
            checks: recentChecks.length,
            timeframe: `${hours}h`
        };
    }
