#!/usr/bin/env node
/**
 * Orko Morning Assistant - Main Agent Controller
 * Autonomous personal assistant for managing morning school drop-off routine
 *
 * Features:
 * - 7:00 AM weather alerts with clothing recommendations
 * - 7:45 AM traffic updates for 8:05 AM departure
 * - Real-time school alerts and delays
 * - Autonomous operation without manual triggers
 * - Agent feed posting for user visibility
 */
 
import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';
 
// Import services
import WeatherService from './services/weather-service.js';
import TrafficService from './services/traffic-service.js';
import SchoolService from './services/school-service.js';
import FeedPostingService from './services/feed-posting-service.js';
import SchedulerService from './services/scheduler-service.js';
import BriefingTemplates from './templates/briefing-templates.js';
 
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
 
class OrkoMorningAssistant {
  constructor() {
    this.agentId = 'orko-morning-assistant';
    this.version = '1.0.0';
    this.config = null;
    this.isInitialized = false;
    this.isRunning = false;
 
    // Service instances
    this.weatherService = null;
    this.trafficService = null;
    this.schoolService = null;
    this.feedPostingService = null;
    this.schedulerService = null;
 
    // Bind methods
    this.initialize = this.initialize.bind(this);
    this.start = this.start.bind(this);
    this.stop = this.stop.bind(this);
  }
 
  async initialize() {
    try {
      console.log('🌅 Initializing Orko Morning Assistant...');
 
      // Load configuration
      await this.loadConfiguration();
 
      // Initialize all services
      await this.initializeServices();
 
      // Ensure data directories exist
      await this.ensureDataDirectories();
 
      this.isInitialized = true;
      console.log('✅ Orko Morning Assistant initialized successfully');
 
      return true;
    } catch (error) {
      console.error('❌ Failed to initialize Orko Morning Assistant:', error);
      return false;
    }
  }
 
  async loadConfiguration() {
    const configPath = path.join(__dirname, 'config', 'agent-config.json');
 
    try {
      const configData = await fs.promises.readFile(configPath, 'utf8');
      this.config = JSON.parse(configData);
      console.log(`📋 Configuration loaded: ${this.config.name} v${this.config.version}`);
    } catch (error) {
      throw new Error(`Failed to load configuration: ${error.message}`);
    }
  }
 
  async initializeServices() {
    try {
      // Initialize weather service
      this.weatherService = new WeatherService(this.config.services.weather);
      console.log('🌤️ Weather service initialized');
 
      // Initialize traffic service
      this.trafficService = new TrafficService(this.config.services.traffic);
      console.log('🚗 Traffic service initialized');
 
      // Initialize school service
      this.schoolService = new SchoolService(this.config.services.school);
      console.log('🏫 School service initialized');
 
      // Initialize feed posting service
      this.feedPostingService = new FeedPostingService(this.config.services.agent_feed);
      console.log('📡 Feed posting service initialized');
 
      // Initialize scheduler service
      this.schedulerService = new SchedulerService(this.config);
      await this.schedulerService.initialize(
        this.weatherService,
        this.trafficService,
        this.schoolService,
        this.feedPostingService
      );
      console.log('⏰ Scheduler service initialized');
 
    } catch (error) {
      throw new Error(`Failed to initialize services: ${error.message}`);
    }
  }
 
  async ensureDataDirectories() {
    const directories = ['data', 'logs'];
    const basePath = path.join(__dirname);
 
    for (const dir of directories) {
      const dirPath = path.join(basePath, dir);
      try {
        await fs.promises.access(dirPath);
      } catch (error) {
        await fs.promises.mkdir(dirPath, { recursive: true });
        console.log(`📁 Created directory: ${dir}`);
      }
    }
  }
 
  async start() {
    if (!this.isInitialized) {
      console.error('❌ Agent not initialized. Call initialize() first.');
      return false;
    }
 
    if (this.isRunning) {
      console.log('⚠️ Agent already running');
      return true;
    }
 
    try {
      console.log('🚀 Starting Orko Morning Assistant autonomous operation...');
 
      // Start the scheduler for autonomous operation
      await this.schedulerService.startScheduler();
 
      this.isRunning = true;
 
      // Post startup message
      await this.postStartupMessage();
 
      console.log('✅ Orko Morning Assistant is now running autonomously');
      console.log('📅 Schedule:');
      console.log('   • 7:00 AM - Weather alerts & clothing recommendations');
      console.log('   • 7:45 AM - Traffic updates & departure prep');
      console.log('   • Continuous - School alerts monitoring');
 
      return true;
    } catch (error) {
      console.error('❌ Failed to start agent:', error);
      return false;
    }
  }
 
  async stop() {
    if (!this.isRunning) {
      console.log('⚠️ Agent not running');
      return true;
    }
 
    try {
      console.log('🛑 Stopping Orko Morning Assistant...');
 
      // Stop the scheduler
      await this.schedulerService.stopScheduler();
 
      this.isRunning = false;
 
      // Post shutdown message
      await this.postShutdownMessage();
 
      console.log('✅ Orko Morning Assistant stopped');
      return true;
    } catch (error) {
      console.error('❌ Error stopping agent:', error);
      return false;
    }
  }
 
  async postStartupMessage() {
    const startupPost = BriefingTemplates.generateQuickAlert(
      'urgent',
      '🌅 Orko Morning Assistant is now active and monitoring for school drop-off',
      'high'
    );
 
    startupPost.title = '🌅 Morning Assistant Active';
    startupPost.contentBody = `## Morning Assistant Started
 
**Status:** Active and monitoring
**Schedule:** Ready for tomorrow's 7:00 AM weather alert
**Coverage:** Weather, traffic, and school alerts
 
Your autonomous morning assistant is now running and will provide:
• 7:00 AM weather updates with clothing recommendations
• 7:45 AM traffic conditions for 8:05 AM departure
• Real-time school delay and closure alerts
 
---
*Orko Morning Assistant started at ${new Date().toLocaleTimeString()}*`;
 
    await this.feedPostingService.postBriefing(startupPost);
  }
 
  async postShutdownMessage() {
    const shutdownPost = BriefingTemplates.generateQuickAlert(
      'urgent',
      'Morning Assistant has been stopped',
      'medium'
    );
 
    shutdownPost.title = '🛑 Morning Assistant Stopped';
    await this.feedPostingService.postBriefing(shutdownPost);
  }
 
  // Manual trigger methods for testing
  async testWeatherAlert() {
    if (!this.isInitialized) {
      throw new Error('Agent not initialized');
    }
 
    console.log('🧪 Testing weather alert...');
    await this.schedulerService.executeWeatherAlert();
  }
 
  async testTrafficUpdate() {
    if (!this.isInitialized) {
      throw new Error('Agent not initialized');
    }
 
    console.log('🧪 Testing traffic update...');
    await this.schedulerService.executeTrafficUpdate();
  }
 
  async testSchoolAlert() {
    if (!this.isInitialized) {
      throw new Error('Agent not initialized');
    }
 
    console.log('🧪 Testing school alert...');
    await this.schedulerService.executeSchoolMonitoring();
  }
 
  // Status and monitoring methods
  getStatus() {
    return {
      agent_id: this.agentId,
      version: this.version,
      initialized: this.isInitialized,
      running: this.isRunning,
      scheduler_status: this.schedulerService ? this.schedulerService.getStatus() : null
    };
  }
 
  async getServiceStatus() {
    if (!this.isInitialized) {
      return { error: 'Agent not initialized' };
    }
 
    return {
      weather: this.weatherService.lastUpdate,
      traffic: this.trafficService.lastUpdate,
      school: this.schoolService.lastUpdate,
      posting_stats: await this.feedPostingService.getPostingStats()
    };
  }
}
 
// Command line interface
if (import.meta.url === `file://${process.argv[1]}`) {
  const agent = new OrkoMorningAssistant();
 
  async function main() {
    const command = process.argv[2] || 'start';
 
    switch (command) {
      case 'start':
        if (await agent.initialize()) {
          await agent.start();
          // Keep process running
          process.on('SIGINT', async () => {
            console.log('\\n📋 Received SIGINT, shutting down gracefully...');
            await agent.stop();
            process.exit(0);
          });
        }
        break;
 
      case 'test-weather':
        if (await agent.initialize()) {
          await agent.testWeatherAlert();
          process.exit(0);
        }
        break;
 
      case 'test-traffic':
        if (await agent.initialize()) {
          await agent.testTrafficUpdate();
          process.exit(0);
        }
        break;
 
      case 'test-school':
        if (await agent.initialize()) {
          await agent.testSchoolAlert();
          process.exit(0);
        }
        break;
 
      case 'status':
        if (await agent.initialize()) {
          const status = agent.getStatus();
          const serviceStatus = await agent.getServiceStatus();
          console.log('📊 Agent Status:', JSON.stringify(status, null, 2));
          console.log('🔧 Service Status:', JSON.stringify(serviceStatus, null, 2));
        }
        break;
 
      case 'stop':
        if (await agent.initialize()) {
          await agent.stop();
        }
        break;
 
      default:
        console.log('📖 Usage:');
        console.log('  node orko-morning-assistant.js [command]');
        console.log('');
        console.log('Commands:');
        console.log('  start           Start autonomous operation (default)');
        console.log('  test-weather    Test weather alert');
        console.log('  test-traffic    Test traffic update');
        console.log('  test-school     Test school alert');
        console.log('  status          Show agent status');
        console.log('  stop            Stop agent');
        break;
    }
  }
 
  main().catch(error => {
    console.error('❌ Fatal error:', error);
    process.exit(1);
  });
}
 
export default OrkoMorningAssistant;
