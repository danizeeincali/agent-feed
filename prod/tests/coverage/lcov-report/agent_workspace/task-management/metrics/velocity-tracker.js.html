/**
 * Task Velocity and Metrics Tracking System
 * Provides analytics for task completion, estimation accuracy, and team velocity
 */
 
class VelocityTracker {
  constructor() {
    this.completedTasks = [];
    this.activeProjects = new Map();
  }
 
  /**
   * Calculate team velocity over specified period
   */
  calculateVelocity(periodDays = 7, tasks = null) {
    const tasksToAnalyze = tasks || this.completedTasks;
    const periodStart = new Date();
    periodStart.setDate(periodStart.getDate() - periodDays);
 
    const recentTasks = tasksToAnalyze.filter(task =>
      task.status === 'completed' &&
      new Date(task.completed_at) >= periodStart
    );
 
    const totalPoints = recentTasks.reduce((sum, task) =>
      sum + (task.impact_score || 1), 0
    );
 
    const totalHours = recentTasks.reduce((sum, task) =>
      sum + (task.actual_hours || task.estimated_hours || 0), 0
    );
 
    return {
      period: periodDays,
      tasksCompleted: recentTasks.length,
      totalPoints,
      totalHours,
      pointsPerDay: totalPoints / periodDays,
      hoursPerDay: totalHours / periodDays,
      avgPointsPerTask: recentTasks.length > 0 ? totalPoints / recentTasks.length : 0,
      tasks: recentTasks
    };
  }
 
  /**
   * Generate burndown chart data for sprint/project
   */
  generateBurndown(tasks, sprintDays = 14, startDate = null) {
    const start = startDate || new Date();
    const totalPoints = tasks.reduce((sum, t) => sum + (t.impact_score || 1), 0);
    const dailyProgress = [];
 
    for (let day = 0; day <= sprintDays; day++) {
      const dayDate = new Date(start);
      dayDate.setDate(start.getDate() + day);
 
      const completedByDay = tasks.filter(task =>
        task.status === 'completed' &&
        new Date(task.completed_at) <= dayDate
      );
 
      const completedPoints = completedByDay.reduce((sum, task) =>
        sum + (task.impact_score || 1), 0
      );
 
      const idealBurndown = totalPoints - (totalPoints / sprintDays * day);
 
      dailyProgress.push({
        day,
        date: dayDate.toISOString().split('T')[0],
        remaining: totalPoints - completedPoints,
        completed: completedPoints,
        ideal: Math.max(0, idealBurndown),
        variance: (totalPoints - completedPoints) - idealBurndown
      });
    }
 
    return {
      totalPoints,
      sprintDays,
      dailyProgress,
      projectedCompletion: this.projectCompletion(dailyProgress),
      burndownRate: this.calculateBurndownRate(dailyProgress)
    };
  }
 
  /**
   * Analyze estimation accuracy
   */
  analyzeEstimationAccuracy(tasks = null) {
    const tasksToAnalyze = (tasks || this.completedTasks).filter(task =>
      task.status === 'completed' &&
      task.estimated_hours &&
      task.actual_hours
    );
 
    if (tasksToAnalyze.length === 0) {
      return { error: 'No completed tasks with both estimated and actual hours' };
    }
 
    const accuracyData = tasksToAnalyze.map(task => {
      const variance = (task.actual_hours - task.estimated_hours) / task.estimated_hours;
      return {
        taskId: task.id,
        title: task.title,
        estimated: task.estimated_hours,
        actual: task.actual_hours,
        variance,
        accuracyScore: 1 - Math.abs(variance)
      };
    });
 
    const totalVariance = accuracyData.reduce((sum, item) => sum + Math.abs(item.variance), 0);
    const avgAccuracy = accuracyData.reduce((sum, item) => sum + item.accuracyScore, 0) / accuracyData.length;
 
    return {
      taskCount: tasksToAnalyze.length,
      avgAccuracy: Math.max(0, avgAccuracy),
      avgVariance: totalVariance / tasksToAnalyze.length,
      overEstimated: accuracyData.filter(item => item.variance < -0.2).length,
      underEstimated: accuracyData.filter(item => item.variance > 0.2).length,
      accurateEstimates: accuracyData.filter(item => Math.abs(item.variance) <= 0.2).length,
      worstEstimates: accuracyData
        .sort((a, b) => Math.abs(b.variance) - Math.abs(a.variance))
        .slice(0, 3),
      bestEstimates: accuracyData
        .sort((a, b) => Math.abs(a.variance) - Math.abs(b.variance))
        .slice(0, 3)
    };
  }
 
  /**
   * Priority distribution analysis
   */
  analyzePriorityDistribution(tasks = null) {
    const tasksToAnalyze = tasks || this.completedTasks;
    const distribution = {};
 
    ['P0', 'P1', 'P2', 'P3', 'P5', 'P8'].forEach(priority => {
      const priorityTasks = tasksToAnalyze.filter(t => t.priority === priority);
      distribution[priority] = {
        count: priorityTasks.length,
        percentage: (priorityTasks.length / tasksToAnalyze.length) * 100,
        avgHours: priorityTasks.length > 0 ?
          priorityTasks.reduce((sum, t) => sum + (t.actual_hours || t.estimated_hours || 0), 0) / priorityTasks.length : 0,
        totalImpact: priorityTasks.reduce((sum, t) => sum + (t.impact_score || 0), 0)
      };
    });
 
    return {
      totalTasks: tasksToAnalyze.length,
      distribution,
      healthCheck: this.assessPriorityHealth(distribution)
    };
  }
 
  /**
   * Generate comprehensive project metrics
   */
  generateProjectMetrics(projectId, tasks = null) {
    const projectTasks = (tasks || this.completedTasks).filter(t =>
      t.metadata?.projectId === projectId
    );
 
    if (projectTasks.length === 0) {
      return { error: `No tasks found for project ${projectId}` };
    }
 
    const completedTasks = projectTasks.filter(t => t.status === 'completed');
    const inProgressTasks = projectTasks.filter(t => t.status === 'in_progress');
    const pendingTasks = projectTasks.filter(t => t.status === 'pending');
 
    return {
      projectId,
      overview: {
        totalTasks: projectTasks.length,
        completed: completedTasks.length,
        inProgress: inProgressTasks.length,
        pending: pendingTasks.length,
        completionRate: (completedTasks.length / projectTasks.length) * 100
      },
      velocity: this.calculateVelocity(7, completedTasks),
      burndown: this.generateBurndown(projectTasks),
      estimation: this.analyzeEstimationAccuracy(completedTasks),
      priorities: this.analyzePriorityDistribution(projectTasks),
      timeline: {
        startDate: Math.min(...projectTasks.map(t => new Date(t.created_at).getTime())),
        estimatedCompletion: this.estimateProjectCompletion(projectTasks),
        actualHours: completedTasks.reduce((sum, t) => sum + (t.actual_hours || 0), 0),
        estimatedHoursRemaining: pendingTasks.concat(inProgressTasks)
          .reduce((sum, t) => sum + (t.estimated_hours || 0), 0)
      }
    };
  }
 
  // Helper methods
  projectCompletion(dailyProgress) {
    const recent = dailyProgress.slice(-3);
    const avgBurnRate = recent.reduce((sum, day, idx, arr) =>
      idx === 0 ? 0 : sum + (arr[idx-1].remaining - day.remaining), 0) / (recent.length - 1);
 
    const lastDay = recent[recent.length - 1];
    return avgBurnRate > 0 ? Math.ceil(lastDay.remaining / avgBurnRate) : Infinity;
  }
 
  calculateBurndownRate(dailyProgress) {
    const workingDays = dailyProgress.filter(day => day.completed > 0);
    if (workingDays.length < 2) return 0;
 
    const firstWorking = workingDays[0];
    const lastWorking = workingDays[workingDays.length - 1];
    const daysDiff = lastWorking.day - firstWorking.day;
 
    return daysDiff > 0 ? (firstWorking.remaining - lastWorking.remaining) / daysDiff : 0;
  }
 
  assessPriorityHealth(distribution) {
    const p0p1Percentage = (distribution.P0.percentage + distribution.P1.percentage);
    const p8Percentage = distribution.P8.percentage;
 
    return {
      healthy: p0p1Percentage < 40 && p8Percentage < 20,
      warnings: [
        ...(p0p1Percentage > 40 ? ['Too many high-priority tasks (>40%)'] : []),
        ...(p8Percentage > 20 ? ['Too many low-priority tasks (>20%)'] : []),
        ...(distribution.P0.percentage > 10 ? ['Too many P0 critical tasks (>10%)'] : [])
      ],
      recommendations: this.generatePriorityRecommendations(distribution)
    };
  }
 
  generatePriorityRecommendations(distribution) {
    const recommendations = [];
 
    if (distribution.P0.percentage > 10) {
      recommendations.push('Review P0 tasks - consider if some could be P1');
    }
    if (distribution.P1.percentage > 30) {
      recommendations.push('Reduce P1 tasks by promoting some to P2');
    }
    if (distribution.P8.percentage > 20) {
      recommendations.push('Archive or delete low-value P8 tasks');
    }
 
    return recommendations;
  }
}
 
module.exports = VelocityTracker;
